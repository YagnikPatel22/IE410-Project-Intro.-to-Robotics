#include <Braccio.h>
#include <Servo.h>

Servo base, shoulder, elbow, wrist_rot, wrist_ver, gripper;

void executeStep(int m1, int m2, int m3, int m4, int m5, int m6) {
  Braccio.ServoMovement(20, m1, m2, m3, m4, m5, m6);
  delay(1000);
}

void setup() {
  Braccio.begin();
}

void loop() {
  executeStep(0, 45, 180, 180, 90, 10);
  executeStep(0, 45, 180, 180, 90, 73);
  executeStep(0, 45, 180, 90, 90, 73);
  executeStep(90, 45, 180, 90, 90, 73);
  executeStep(90, 45, 180, 180, 90, 73);
  executeStep(90, 45, 180, 180, 90, 10);
}
