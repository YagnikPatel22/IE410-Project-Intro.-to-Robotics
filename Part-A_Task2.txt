#include <Braccio.h>
#include <Servo.h>

Servo base, shoulder, elbow, wrist_rot, wrist_ver, gripper;

// --- CONFIGURATION ---
const int IR_SENSOR_PIN = A0; 
float Kp_Base = 0.03;         
float Kp_Reach = 0.02;        

// --- POINT B: PREDEFINED DROP LOCATION ---
// Change these specific angles to set exactly where you want the object placed
int drop_base = 180;       // 180 swings the arm to the opposite side
int drop_shoulder = 90;
int drop_elbow = 90;
int drop_wrist_ver = 90;
int drop_wrist_rot = 90;

// Current robot state (Hovering above the workspace)
float current_base = 90;
float current_shoulder = 120; 
float current_elbow = 90;

void setup() {
  Serial.begin(115200);
  pinMode(IR_SENSOR_PIN, INPUT_PULLUP);
  Braccio.begin();
  
  // Go to starting hover position (gripper open: 10 degrees)
  Braccio.ServoMovement(20, current_base, current_shoulder, current_elbow, 90, 90, 10);
  delay(1000);
  
  Serial.println("READY");
}

void loop() {
  if (Serial.available() > 0) {
    String data = Serial.readStringUntil('\n');
    
    if (data == "<GRASP>") {
      executeTactileGrasp();
      Serial.println("READY"); 
    } 
    else if (data.startsWith("<") && data.endsWith(">")) {
      data = data.substring(1, data.length() - 1);
      int commaIndex = data.indexOf(',');
      
      float err_x = data.substring(0, commaIndex).toFloat();
      float err_y = data.substring(commaIndex + 1).toFloat();
      
      nudgeRobot(err_x, err_y);
    }
  }
}

void nudgeRobot(float err_x, float err_y) {
  float base_nudge = err_x * Kp_Base;
  float reach_nudge = err_y * Kp_Reach;

  current_base += base_nudge;
  current_shoulder += reach_nudge; 
  current_elbow -= reach_nudge;    

  current_base = constrain(current_base, 0, 180);
  current_shoulder = constrain(current_shoulder, 45, 135);
  current_elbow = constrain(current_elbow, 15, 165);

  Braccio.ServoMovement(10, (int)current_base, (int)current_shoulder, (int)current_elbow, 90, 90, 10);
  
  delay(100); 
  Serial.println("READY");
}

void executeTactileGrasp() {
  // PHASE 1: STRIKE (Lower arm to the object)
  Braccio.ServoMovement(20, current_base, current_shoulder - 30, current_elbow + 10, 90, 90, 10);
  delay(800);
  
  // PHASE 2: CATCH (Check IR sensor)
  int beamState = digitalRead(IR_SENSOR_PIN);
  
  if (beamState == LOW) {
    // Beam broken! Object is secured. Close gripper (73 degrees).
    Braccio.ServoMovement(20, current_base, current_shoulder - 30, current_elbow + 10, 90, 90, 73);
    delay(800);
    
    // PHASE 3: LIFT 
    Braccio.ServoMovement(20, current_base, 120, 90, 90, 90, 73);
    delay(800);
    
    // PHASE 4: TRANSIT TO PREDEFINED LOCATION (Point B)
    Braccio.ServoMovement(20, drop_base, drop_shoulder, drop_elbow, drop_wrist_ver, drop_wrist_rot, 73);
    delay(1000);
    
    // PHASE 5: RELEASE
    Braccio.ServoMovement(20, drop_base, drop_shoulder, drop_elbow, drop_wrist_ver, drop_wrist_rot, 10);
    delay(800);
  } else {
    // Grasp failed (Object missed). Do not execute drop-off sequence.
  }
  
  // PHASE 6: RETURN TO HOVER (Ready for the next task)
  current_base = 90;
  current_shoulder = 120;
  current_elbow = 90;
  Braccio.ServoMovement(20, current_base, current_shoulder, current_elbow, 90, 90, 10);
  delay(1000);
}
